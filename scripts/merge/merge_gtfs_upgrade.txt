CREATE OR REPLACE FUNCTION merge_gtfs_schemas_with_remap(
    source_schema TEXT,
    target_schema TEXT
) RETURNS JSONB AS $$
DECLARE
    result JSONB;

    -- таблица_имя:колонка_id, для которых нужно сгенерировать новые идентификаторы
    mapping_tables TEXT[] := ARRAY[
        'agency:agency_id',
        'routes:route_id',
        'stops:stop_id',
        'calendar:service_id',
        'shapes:shape_id',
        'trips:trip_id'
    ];

    table_name      TEXT;
    id_column       TEXT;
    table_exists    BOOLEAN;

    deleted_agencies_count INTEGER := 0;
    migrated_count         BIGINT  := 0;
    total_migrated         BIGINT  := 0;

    max_shapes_id       INTEGER := 0;
    max_frequencies_row INTEGER := 0;
    max_transfers_id    INTEGER := 0;
BEGIN

    -- БАЗОВЫЕ ПРОВЕРКИ
    ------------------------------------------------------------------------
    IF source_schema IS NULL OR target_schema IS NULL THEN
        RAISE EXCEPTION 'source_schema и target_schema не должны быть NULL';
    END IF;

    IF source_schema = target_schema THEN
        RAISE EXCEPTION 'source_schema и target_schema не должны совпадать (%). Ожидается миграция между разными схемами.', source_schema;
    END IF;


    -- ПРЕДВАРИТЕЛЬНЫЕ МАКСИМУМЫ ДЛЯ SURROGATE KEY В ТАРГЕТЕ
    ------------------------------------------------------------------------
    SELECT to_regclass(format('%I.%I', target_schema, 'shapes')) IS NOT NULL
      INTO table_exists;
    IF table_exists THEN
        EXECUTE format('SELECT COALESCE(MAX(id), 0) FROM %I.shapes', target_schema)
           INTO max_shapes_id;
    END IF;

    SELECT to_regclass(format('%I.%I', target_schema, 'frequencies')) IS NOT NULL
      INTO table_exists;
    IF table_exists THEN
        EXECUTE format('SELECT COALESCE(MAX(frequencies_row), 0) FROM %I.frequencies', target_schema)
           INTO max_frequencies_row;
    END IF;

    SELECT to_regclass(format('%I.%I', target_schema, 'transfers')) IS NOT NULL
      INTO table_exists;
    IF table_exists THEN
        EXECUTE format('SELECT COALESCE(MAX(id), 0) FROM %I.transfers', target_schema)
           INTO max_transfers_id;
    END IF;


    -- ВСПОМОГАТЕЛЬНЫЕ ВРЕМЕННЫЕ ТАБЛИЦЫ
    -- TEMP таблицы помечены ON COMMIT DROP (теперь не чистим вручную)
    ------------------------------------------------------------------------
    CREATE TEMP TABLE IF NOT EXISTS id_mapping (
        table_name TEXT,
        old_id     TEXT,
        new_id     TEXT,
        PRIMARY KEY (table_name, old_id)
    ) ON COMMIT DROP;

    CREATE TEMP TABLE IF NOT EXISTS duplicate_agencies (
        agency_id   TEXT,
        agency_name TEXT,
        PRIMARY KEY (agency_id)
    ) ON COMMIT DROP;


    -- Этап 1. УДАЛЕНИЕ ДУБЛЕЙ АГЕНТСТВ В TARGET ПО ИМЕНИ
    -- Аккуратно удаляем все зависимые строки в правильном порядке.
    ------------------------------------------------------------------------
    SELECT to_regclass(format('%I.%I', source_schema, 'agency')) IS NOT NULL
      INTO table_exists;
    IF table_exists THEN
        SELECT to_regclass(format('%I.%I', target_schema, 'agency')) IS NOT NULL
          INTO table_exists;
        IF table_exists THEN
            -- Список агентств в target, у которых имя совпадает с агентством из source
            EXECUTE format(
                'INSERT INTO duplicate_agencies (agency_id, agency_name)
                 SELECT t.agency_id, t.agency_name
                 FROM %I.agency t
                 WHERE EXISTS (
                     SELECT 1
                     FROM %I.agency s
                     WHERE s.agency_name = t.agency_name
                 )
                 ON CONFLICT (agency_id) DO NOTHING',
                target_schema, source_schema
            );

            -- Удаляем сразу всеми батчами, а не по одному агентству в цикле.
            -- transfers
            EXECUTE format(
                'DELETE FROM %I.transfers tr
                 USING %I.routes r, duplicate_agencies da
                 WHERE (
                         tr.from_route_id = r.route_id
                      OR tr.to_route_id   = r.route_id
                 )
                 AND r.agency_id = da.agency_id',
                target_schema, target_schema
            );

            EXECUTE format(
                'DELETE FROM %I.transfers tr
                 USING %I.trips t, %I.routes r, duplicate_agencies da
                 WHERE (
                         tr.from_trip_id = t.trip_id
                      OR tr.to_trip_id   = t.trip_id
                 )
                 AND t.route_id = r.route_id
                 AND r.agency_id = da.agency_id',
                target_schema, target_schema, target_schema
            );

            -- stop_times
            EXECUTE format(
                'DELETE FROM %I.stop_times st
                 USING %I.trips t, %I.routes r, duplicate_agencies da
                 WHERE st.trip_id = t.trip_id
                 AND t.route_id = r.route_id
                 AND r.agency_id = da.agency_id',
                target_schema, target_schema, target_schema
            );

            -- frequencies
            EXECUTE format(
                'DELETE FROM %I.frequencies f
                 USING %I.trips t, %I.routes r, duplicate_agencies da
                 WHERE f.trip_id = t.trip_id
                 AND t.route_id = r.route_id
                 AND r.agency_id = da.agency_id',
                target_schema, target_schema, target_schema
            );

            -- trips
            EXECUTE format(
                'DELETE FROM %I.trips t
                 USING %I.routes r, duplicate_agencies da
                 WHERE t.route_id = r.route_id
                 AND r.agency_id = da.agency_id',
                target_schema, target_schema
            );

            -- routes
            EXECUTE format(
                'DELETE FROM %I.routes r
                 USING duplicate_agencies da
                 WHERE r.agency_id = da.agency_id',
                target_schema
            );

            -- agency
            EXECUTE format(
                'DELETE FROM %I.agency a
                 USING duplicate_agencies da
                 WHERE a.agency_id = da.agency_id',
                target_schema
            );

            GET DIAGNOSTICS deleted_agencies_count = ROW_COUNT;

            RAISE NOTICE 'Удалено % дублирующихся агентств из target схемы', deleted_agencies_count;
        END IF;
    END IF;


    -- Этап 2. ГЕНЕРАЦИЯ ID-MAPPING ДЛЯ ВСЕХ ПОДДЕРЖИВАЕМЫХ ТАБЛИЦ
    -- (детерминированный префикс + md5 + timestamp для минимизации коллизий)
    ------------------------------------------------------------------------
    FOR i IN 1..array_length(mapping_tables, 1) LOOP
        table_name := split_part(mapping_tables[i], ':', 1);
        id_column  := split_part(mapping_tables[i], ':', 2);

        SELECT to_regclass(format('%I.%I', source_schema, table_name)) IS NOT NULL
          INTO table_exists;

        IF NOT table_exists THEN
            RAISE NOTICE 'Таблица %.% не существует, пропускаем', source_schema, table_name;
            CONTINUE;
        END IF;

        EXECUTE format(
            'INSERT INTO id_mapping (table_name, old_id, new_id)
             SELECT $1, s.%I,
                    substr($1, 1, 2) || ''_'' ||
                    substr(md5($1 || s.%I || random()::text), 1, 10) || ''_'' ||
                    round(extract(epoch from clock_timestamp())*1000)::text
             FROM %I.%I s
             ON CONFLICT DO NOTHING',
            id_column, id_column, source_schema, table_name
        ) USING table_name;

        RAISE NOTICE 'Создан маппинг для таблицы %', table_name;
    END LOOP;


    -- Этап 3. ПЕРЕНОС ДАННЫХ С УЧЁТОМ МАППИНГОВ
    -- Порядок вставки выбран так, чтобы соблюдать внешние ключи.
    ------------------------------------------------------------------------
    -- agency
    SELECT to_regclass(format('%I.%I', source_schema, 'agency')) IS NOT NULL
      INTO table_exists;
    IF table_exists THEN
        SELECT to_regclass(format('%I.%I', target_schema, 'agency')) IS NOT NULL
          INTO table_exists;
        IF table_exists THEN
            EXECUTE format(
                'INSERT INTO %I.agency (
                     agency_id,
                     agency_name,
                     agency_url,
                     agency_timezone,
                     agency_lang,
                     agency_phone,
                     agency_fare_url,
                     agency_email
                 )
                 SELECT
                     m.new_id AS agency_id,
                     s.agency_name,
                     s.agency_url,
                     s.agency_timezone,
                     s.agency_lang,
                     s.agency_phone,
                     s.agency_fare_url,
                     s.agency_email
                 FROM %I.agency s
                 JOIN id_mapping m
                   ON m.table_name = ''agency'' AND m.old_id = s.agency_id',
                target_schema, source_schema
            );
            GET DIAGNOSTICS migrated_count = ROW_COUNT;
            total_migrated := total_migrated + migrated_count;
            RAISE NOTICE 'agency: перенесено % записей', migrated_count;
        END IF;
    END IF;

    -- routes
    SELECT to_regclass(format('%I.%I', source_schema, 'routes')) IS NOT NULL
      INTO table_exists;
    IF table_exists THEN
        SELECT to_regclass(format('%I.%I', target_schema, 'routes')) IS NOT NULL
          INTO table_exists;
        IF table_exists THEN
            EXECUTE format(
                'INSERT INTO %I.routes (
                     route_id,
                     agency_id,
                     route_short_name,
                     route_long_name,
                     route_desc,
                     route_type,
                     route_url,
                     route_color,
                     route_text_color,
                     route_sort_order
                 )
                 SELECT
                     m_routes.new_id AS route_id,
                     m_agency.new_id AS agency_id,
                     s.route_short_name,
                     s.route_long_name,
                     s.route_desc,
                     s.route_type::text::%I.route_type_val,
                     s.route_url,
                     s.route_color,
                     s.route_text_color,
                     s.route_sort_order
                 FROM %I.routes s
                 JOIN id_mapping m_routes
                   ON m_routes.table_name = ''routes'' AND m_routes.old_id = s.route_id
                 LEFT JOIN id_mapping m_agency
                   ON m_agency.table_name = ''agency'' AND m_agency.old_id = s.agency_id',
                target_schema, target_schema, source_schema
            );
            GET DIAGNOSTICS migrated_count = ROW_COUNT;
            total_migrated := total_migrated + migrated_count;
            RAISE NOTICE 'routes: перенесено % записей', migrated_count;
        END IF;
    END IF;

    -- stops
    SELECT to_regclass(format('%I.%I', source_schema, 'stops')) IS NOT NULL
      INTO table_exists;
    IF table_exists THEN
        SELECT to_regclass(format('%I.%I', target_schema, 'stops')) IS NOT NULL
          INTO table_exists;
        IF table_exists THEN
            EXECUTE format(
                'INSERT INTO %I.stops (
                     stop_id,
                     stop_code,
                     stop_name,
                     stop_desc,
                     stop_loc,
                     zone_id,
                     stop_url,
                     location_type,
                     parent_station,
                     stop_timezone,
                     wheelchair_boarding,
                     level_id,
                     platform_code
                 )
                 SELECT
                     m.new_id AS stop_id,
                     s.stop_code,
                     s.stop_name,
                     s.stop_desc,
                     s.stop_loc,
                     s.zone_id,
                     s.stop_url,
                     s.location_type::text::%I.location_type_val,
                     m_parent.new_id AS parent_station,
                     s.stop_timezone,
                     s.wheelchair_boarding::text::%I.wheelchair_boarding_val,
                     s.level_id,
                     s.platform_code
                 FROM %I.stops s
                 JOIN id_mapping m
                   ON m.table_name = ''stops'' AND m.old_id = s.stop_id
                 LEFT JOIN id_mapping m_parent
                   ON m_parent.table_name = ''stops'' AND m_parent.old_id = s.parent_station',
                target_schema, target_schema, target_schema, source_schema
            );
            GET DIAGNOSTICS migrated_count = ROW_COUNT;
            total_migrated := total_migrated + migrated_count;
            RAISE NOTICE 'stops: перенесено % записей', migrated_count;
        END IF;
    END IF;

    -- calendar
    SELECT to_regclass(format('%I.%I', source_schema, 'calendar')) IS NOT NULL
      INTO table_exists;
    IF table_exists THEN
        SELECT to_regclass(format('%I.%I', target_schema, 'calendar')) IS NOT NULL
          INTO table_exists;
        IF table_exists THEN
            EXECUTE format(
                'INSERT INTO %I.calendar (
                     service_id,
                     monday,
                     tuesday,
                     wednesday,
                     thursday,
                     friday,
                     saturday,
                     sunday,
                     start_date,
                     end_date
                 )
                 SELECT
                     m.new_id AS service_id,
                     s.monday::text::%I.availability,
                     s.tuesday::text::%I.availability,
                     s.wednesday::text::%I.availability,
                     s.thursday::text::%I.availability,
                     s.friday::text::%I.availability,
                     s.saturday::text::%I.availability,
                     s.sunday::text::%I.availability,
                     s.start_date,
                     s.end_date
                 FROM %I.calendar s
                 JOIN id_mapping m
                   ON m.table_name = ''calendar'' AND m.old_id = s.service_id',
                target_schema,
                target_schema, target_schema, target_schema, target_schema,
                target_schema, target_schema, target_schema,
                source_schema
            );
            GET DIAGNOSTICS migrated_count = ROW_COUNT;
            total_migrated := total_migrated + migrated_count;
            RAISE NOTICE 'calendar: перенесено % записей', migrated_count;
        END IF;
    END IF;

    -- calendar_dates
    SELECT to_regclass(format('%I.%I', source_schema, 'calendar_dates')) IS NOT NULL
      INTO table_exists;
    IF table_exists THEN
        SELECT to_regclass(format('%I.%I', target_schema, 'calendar_dates')) IS NOT NULL
          INTO table_exists;
        IF table_exists THEN
            EXECUTE format(
                'INSERT INTO %I.calendar_dates (
                     service_id,
                     date,
                     exception_type
                 )
                 SELECT
                     m.new_id AS service_id,
                     s.date,
                     s.exception_type::text::%I.exception_type_v
                 FROM %I.calendar_dates s
                 JOIN id_mapping m
                   ON m.table_name = ''calendar'' AND m.old_id = s.service_id',
                target_schema, target_schema, source_schema
            );
            GET DIAGNOSTICS migrated_count = ROW_COUNT;
            total_migrated := total_migrated + migrated_count;
            RAISE NOTICE 'calendar_dates: перенесено % записей', migrated_count;
        END IF;
    END IF;

    -- shapes
    SELECT to_regclass(format('%I.%I', source_schema, 'shapes')) IS NOT NULL
      INTO table_exists;
    IF table_exists THEN
        SELECT to_regclass(format('%I.%I', target_schema, 'shapes')) IS NOT NULL
          INTO table_exists;
        IF table_exists THEN
            EXECUTE format(
                'INSERT INTO %I.shapes (
                     id,
                     shape_id,
                     shape_pt_sequence,
                     shape_pt_loc,
                     shape_dist_traveled
                 )
                 SELECT
                     row_number() OVER () + $1 AS id,
                     m.new_id AS shape_id,
                     s.shape_pt_sequence,
                     s.shape_pt_loc,
                     s.shape_dist_traveled
                 FROM %I.shapes s
                 JOIN id_mapping m
                   ON m.table_name = ''shapes'' AND m.old_id = s.shape_id',
                target_schema, source_schema
            ) USING max_shapes_id;
            GET DIAGNOSTICS migrated_count = ROW_COUNT;
            total_migrated := total_migrated + migrated_count;
            RAISE NOTICE 'shapes: перенесено % записей', migrated_count;
        END IF;
    END IF;

    -- trips
    SELECT to_regclass(format('%I.%I', source_schema, 'trips')) IS NOT NULL
      INTO table_exists;
    IF table_exists THEN
        SELECT to_regclass(format('%I.%I', target_schema, 'trips')) IS NOT NULL
          INTO table_exists;
        IF table_exists THEN
            EXECUTE format(
                'INSERT INTO %I.trips (
                     trip_id,
                     route_id,
                     service_id,
                     trip_headsign,
                     trip_short_name,
                     direction_id,
                     block_id,
                     shape_id,
                     wheelchair_accessible,
                     bikes_allowed
                 )
                 SELECT
                     m_trips.new_id AS trip_id,
                     m_routes.new_id AS route_id,
                     m_calendar.new_id AS service_id,
                     s.trip_headsign,
                     s.trip_short_name,
                     s.direction_id,
                     s.block_id,
                     m_shapes.new_id AS shape_id,
                     s.wheelchair_accessible::text::%I.wheelchair_accessibility,
                     s.bikes_allowed::text::%I.bikes_allowance
                 FROM %I.trips s
                 JOIN id_mapping m_trips
                   ON m_trips.table_name = ''trips'' AND m_trips.old_id = s.trip_id
                 JOIN id_mapping m_routes
                   ON m_routes.table_name = ''routes'' AND m_routes.old_id = s.route_id
                 JOIN id_mapping m_calendar
                   ON m_calendar.table_name = ''calendar'' AND m_calendar.old_id = s.service_id
                 LEFT JOIN id_mapping m_shapes
                   ON m_shapes.table_name = ''shapes'' AND m_shapes.old_id = s.shape_id',
                target_schema, target_schema, target_schema, source_schema
            );
            GET DIAGNOSTICS migrated_count = ROW_COUNT;
            total_migrated := total_migrated + migrated_count;
            RAISE NOTICE 'trips: перенесено % записей', migrated_count;
        END IF;
    END IF;

    -- stop_times
    SELECT to_regclass(format('%I.%I', source_schema, 'stop_times')) IS NOT NULL
      INTO table_exists;
    IF table_exists THEN
        SELECT to_regclass(format('%I.%I', target_schema, 'stop_times')) IS NOT NULL
          INTO table_exists;
        IF table_exists THEN
            EXECUTE format(
                'INSERT INTO %I.stop_times (
                     trip_id,
                     arrival_time,
                     departure_time,
                     stop_id,
                     stop_sequence,
                     stop_sequence_consec,
                     stop_headsign,
                     pickup_type,
                     drop_off_type,
                     shape_dist_traveled,
                     timepoint,
                     trip_start_time
                 )
                 SELECT
                     m_trips.new_id AS trip_id,
                     s.arrival_time,
                     s.departure_time,
                     m_stops.new_id AS stop_id,
                     s.stop_sequence,
                     s.stop_sequence_consec,
                     s.stop_headsign,
                     s.pickup_type::text::%I.pickup_drop_off_type,
                     s.drop_off_type::text::%I.pickup_drop_off_type,
                     s.shape_dist_traveled,
                     s.timepoint::text::%I.timepoint_v,
                     s.trip_start_time
                 FROM %I.stop_times s
                 JOIN id_mapping m_trips
                   ON m_trips.table_name = ''trips'' AND m_trips.old_id = s.trip_id
                 JOIN id_mapping m_stops
                   ON m_stops.table_name = ''stops'' AND m_stops.old_id = s.stop_id',
                target_schema, target_schema, target_schema, target_schema, source_schema
            );
            GET DIAGNOSTICS migrated_count = ROW_COUNT;
            total_migrated := total_migrated + migrated_count;
            RAISE NOTICE 'stop_times: перенесено % записей', migrated_count;
        END IF;
    END IF;

    -- frequencies
    SELECT to_regclass(format('%I.%I', source_schema, 'frequencies')) IS NOT NULL
      INTO table_exists;
    IF table_exists THEN
        SELECT to_regclass(format('%I.%I', target_schema, 'frequencies')) IS NOT NULL
          INTO table_exists;
        IF table_exists THEN
            EXECUTE format(
                'INSERT INTO %I.frequencies (
                     frequencies_row,
                     trip_id,
                     start_time,
                     end_time,
                     headway_secs,
                     exact_times
                 )
                 SELECT
                     row_number() OVER () + $1 AS frequencies_row,
                     m_trips.new_id AS trip_id,
                     s.start_time,
                     s.end_time,
                     s.headway_secs,
                     s.exact_times::text::%I.exact_times_v
                 FROM %I.frequencies s
                 JOIN id_mapping m_trips
                   ON m_trips.table_name = ''trips'' AND m_trips.old_id = s.trip_id',
                target_schema, target_schema, source_schema
            ) USING max_frequencies_row;
            GET DIAGNOSTICS migrated_count = ROW_COUNT;
            total_migrated := total_migrated + migrated_count;
            RAISE NOTICE 'frequencies: перенесено % записей', migrated_count;
        END IF;
    END IF;

    -- transfers
    SELECT to_regclass(format('%I.%I', source_schema, 'transfers')) IS NOT NULL
      INTO table_exists;
    IF table_exists THEN
        SELECT to_regclass(format('%I.%I', target_schema, 'transfers')) IS NOT NULL
          INTO table_exists;
        IF table_exists THEN
            EXECUTE format(
                'INSERT INTO %I.transfers (
                     id,
                     from_stop_id,
                     to_stop_id,
                     transfer_type,
                     min_transfer_time,
                     from_route_id,
                     to_route_id,
                     from_trip_id,
                     to_trip_id
                 )
                 SELECT
                     row_number() OVER () + $1 AS id,
                     m_from_stop.new_id AS from_stop_id,
                     m_to_stop.new_id   AS to_stop_id,
                     s.transfer_type::text::%I.transfer_type_v,
                     s.min_transfer_time,
                     m_from_route.new_id AS from_route_id,
                     m_to_route.new_id   AS to_route_id,
                     m_from_trip.new_id  AS from_trip_id,
                     m_to_trip.new_id    AS to_trip_id
                 FROM %I.transfers s
                 LEFT JOIN id_mapping m_from_stop
                   ON m_from_stop.table_name = ''stops'' AND m_from_stop.old_id = s.from_stop_id
                 LEFT JOIN id_mapping m_to_stop
                   ON m_to_stop.table_name = ''stops'' AND m_to_stop.old_id = s.to_stop_id
                 LEFT JOIN id_mapping m_from_route
                   ON m_from_route.table_name = ''routes'' AND m_from_route.old_id = s.from_route_id
                 LEFT JOIN id_mapping m_to_route
                   ON m_to_route.table_name = ''routes'' AND m_to_route.old_id = s.to_route_id
                 LEFT JOIN id_mapping m_from_trip
                   ON m_from_trip.table_name = ''trips'' AND m_from_trip.old_id = s.from_trip_id
                 LEFT JOIN id_mapping m_to_trip
                   ON m_to_trip.table_name = ''trips'' AND m_to_trip.old_id = s.to_trip_id',
                target_schema, target_schema, source_schema
            ) USING max_transfers_id;
            GET DIAGNOSTICS migrated_count = ROW_COUNT;
            total_migrated := total_migrated + migrated_count;
            RAISE NOTICE 'transfers: перенесено % записей', migrated_count;
        END IF;
    END IF;

    -- feed_info
    SELECT to_regclass(format('%I.%I', source_schema, 'feed_info')) IS NOT NULL
      INTO table_exists;
    IF table_exists THEN
        SELECT to_regclass(format('%I.%I', target_schema, 'feed_info')) IS NOT NULL
          INTO table_exists;
        IF table_exists THEN
            EXECUTE format(
                'INSERT INTO %I.feed_info
                 SELECT *
                 FROM %I.feed_info
                 ON CONFLICT DO NOTHING',
                target_schema, source_schema
            );
            GET DIAGNOSTICS migrated_count = ROW_COUNT;
            total_migrated := total_migrated + migrated_count;
            RAISE NOTICE 'feed_info: перенесено % записей', migrated_count;
        END IF;
    END IF;

    ------------------------------------------------------------------------
    -- ФИНАЛЬНЫЙ JSON-РЕЗУЛЬТАТ
    ------------------------------------------------------------------------
    SELECT jsonb_build_object(
               'status', 'success',
               'total_rows_migrated', total_migrated,
               'duplicate_agencies_removed', deleted_agencies_count,
               'tables_processed', 11,
               'id_mappings_created', (SELECT COUNT(*) FROM id_mapping),
               'source_schema', source_schema,
               'target_schema', target_schema
           )
      INTO result;

    RETURN result;

EXCEPTION
    WHEN OTHERS THEN
        -- В случае ошибки возвращаем структурированное описание, чтобы система
        -- мониторинга могла его залогировать и отобразить пользователю.
        RETURN jsonb_build_object(
            'status', 'error',
            'error_message', SQLERRM,
            'error_detail', SQLSTATE,
            'source_schema', source_schema,
            'target_schema', target_schema
        );
END;
$$ LANGUAGE plpgsql;
